  • Enter a string from user and convert into title case 
    #include<stdio.h>
#include<conio.h>
#inlude<dos.h>
void main()
 {
char str[100];
int i;
printf("Enter a string: ");
gets(str);
// convert first letter to uppercase (if it is lowercase)
if(str[0] >= 'a' && str[0] <= 'z')
str[0] = str[0] - 32;
for(i = 1; str[i] != '\0'; i++) {
// if previous char is space, make current char uppercase
if(str[i - 1] == ' ' && str[i] >= 'a' && str[i] <= 'z')
str[i] = str[i] - 32;
// else make it lowercase
else if(str[i - 1] != ' ' && str[i] >= 'A' && str[i] <= 'Z')
str[i] = str[i] + 32;
}
printf("Title case: %s", str);
return 0;
}

  • string palindrome example(using reverse) 
    #include<stdio.h>
#include<conio.h>
#include<dos.h>
void main()
{
char str[100], rev[100];
int i, j, len;
clrscr();
printf("Enter a string: ");
gets(str);  // use fgets in modern compilers
len = strlen(str);
// reverse the string
for(i = len - 1, j = 0; i >= 0; i--, j++)
{
rev[j] = str[i];
}
rev[j] = '\0';  // end reversed string
// compare both strings
if(strcmp(str, rev) == 0)
printf("String is palindrome.");
else
printf("String is not palindrome.");
return 0;
}
