struct Employee
{

char designation[50];
};

{
printf("\nEnter details of employee %d:\n", i + 1);

}
printf("\n---------------------------------------------------------\n");
printf("Emp.ID\t\tName\t\tDesignation\n");

{
printf("%d\t\t%-10s\t%-10s\n", emp[i].id, emp[i].name, emp[i].designation);
}
printf("---------------------------------------------------------\n");
return 0;
}

struct Employee
{
int id;
char name[50];
char designation[50];
};
int main()
{
struct Employee emp[50];
int i, j, n;

scanf("%d", &n);
for(i = 0; i < n; i++)
{
printf("\nEnter details of employee %d:\n", i + 1);
printf("Enter Employee ID: ");
scanf("%d", &emp[i].id);
printf("Enter Name: ");
scanf("%s", emp[i].name);   // use fgets for full name
printf("Enter Designation: ");
scanf("%s", emp[i].designation);
}
printf("\n-------------------------------------------\n");
printf("Emp.ID\t\tName (character by character)\n");
printf("-------------------------------------------\n");
for(i = 0; i < n; i++)
{
printf("%d\t\t", emp[i].id);
for(j = 0; emp[i].name[j] != '\0'; j++)
{
printf("%c ", emp[i].name[j]);  // print each character with space
}
printf("\n");
}
printf("-------------------------------------------\n");
return 0;
}
  • enter student details and print all details by ascending order of roll number 
    #include<stdio.h>
#include<conio.h>
#inlude<dos.h>
struct Student
{
int roll;
char name[50];
float fee;
char dob[15];  // date of birth as string (e.g., 12-05-2004)
};
int main()
{
struct Student s[50], temp;
int i, j, n;
printf("Enter number of students (max 50): ");
scanf("%d", &n);
for(i = 0; i < n; i++)
{
printf("\nEnter details of student %d:\n", i + 1);
printf("Enter Roll No: ");
scanf("%d", &s[i].roll);
printf("Enter Name: ");
scanf("%s", s[i].name);  // use fgets if full name needed
printf("Enter Fee: ");
scanf("%f", &s[i].fee);
printf("Enter DOB (dd-mm-yyyy): ");
scanf("%s", s[i].dob);
}
// Sorting by roll number (ascending)
for(i = 0; i < n - 1; i++)
{
for(j = i + 1; j < n; j++)
{
if(s[i].roll > s[j].roll)
{
temp = s[i];
s[i] = s[j];
s[j] = temp;
}
}
}
printf("\n-----------------------------------------------\n");
printf("Roll\tName\t\tFee\t\tDOB\n");
printf("-----------------------------------------------\n");
for(i = 0; i < n; i++)
{
printf("%d\t%-10s\t%.2f\t\t%s\n", s[i].roll, s[i].name, s[i].fee, s[i].dob);
}
printf("-----------------------------------------------\n");
return 0;
}
