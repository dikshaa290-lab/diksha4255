  • Print all ASCII character 0 to 127 ? 
    #include<stdio.h>
#include<conio.h>
#include<dos.h>
int main()
{
int i;
printf("ASCII characters from 0 to 127:\n\n");
for(i=0;i<=127;i++)
{
printf("character for ASCII %d is: ",i);
if(i>=32&& i<=126)
{
printf("%c\n", i);
}
else
{
printf("Non-printable\n");
}
}
return 0;
}
output:
ASCII characters from 0 to 127:

0-31    : Non-printable
32-126  : Printable characters
127     : Non-printable
