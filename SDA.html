  • printing array in ascending order 
    #include<stdio.h>
#include<conio.h>
#include<dos.h>
void main() {
    int arr[5], i, j, temp;
    printf("Enter 5 elements:\n");
    for(i = 0; i < 5; i++)
        scanf("%d", &arr[i]);

    for(i = 0; i < 5; i++) {
        for(j = i + 1; j < 5; j++) {
            if(arr[i] > arr[j]) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }

    printf("Array in ascending order:\n");
    for(i = 0; i < 5; i++)
        printf("%d ", arr[i]);
}
