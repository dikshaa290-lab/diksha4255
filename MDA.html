  • storing and printing transpose of matrix 
    #include<stdio.h>
#include<conio.h>
#inlude<dos.h>
void main() {
    int arr[3][3], brr[3][3], i, j;
    clrscr();
    printf("Enter elements for 3X3 matrix:\n");
    for(i = 0; i < 3; i++) {
        for(j = 0; j < 3; j++) {
            scanf("%d", &arr[i][j]);
        }
    }
    for(i = 0; i < 3; i++) {
        for(j = 0; j < 3; j++) {
            brr[j][i] = arr[i][j];
        }
    }
    printf("\nTranspose of matrix:\n");
    for(i = 0; i < 3; i++) {
        for(j = 0; j < 3; j++) {
            printf("%d\t", brr[i][j]);
        }
        printf("\n");
    }
}

#include<conio.h>

    for(i = 0; i < 3; i++)
        for(j = 0; j < 3; j++)
            scanf("%d", &arr[i][j]);
    printf("Enter second 3x3 matrix:\n");
    for(i = 0; i < 3; i++)
        for(j = 0; j < 3; j++)
            scanf("%d", &brr[i][j]);
    for(i = 0; i < 3; i++)
        for(j = 0; j < 3; j++)

    for(i = 0; i < 3; i++) {
        for(j = 0; j < 3; j++)
            printf("%d\t", sum[i][j]);
        printf("\n");
    }
}
