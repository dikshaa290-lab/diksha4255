  • swap numbers using call by address? 
    
#include<stdio.h>
#include<conio.h>
#include<dos.h>
void swap(int *a, int *b)
{
int temp= *a;
*a = *b;
*b = temp;
}
int main()
{
int x=5, y=10;
printf("Before swap: x= %d, y%d\n", x, y);
swap(&x, &y);
printf("After swap: x = %d, y= %d\n", x, y);
return 0;
}
output:

Swapped lists:[20,10]

{
int temp = a;
a=b;
b=temp;
printf("Inside function: a= %d, b= %d\n" , a, b);
}
int main()
{

printf("Before swap : x=%d, y=%d\n", x,y);
